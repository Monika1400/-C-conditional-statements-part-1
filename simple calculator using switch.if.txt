#include<stdio.h>
int main() {
    char operator;
    scanf("%c",&operator);
    int a,b;
    scanf("%d %d",&a,&b);
    switch (operator) {
        case'+':
            printf("%d",a+b);
            break;
        case'-':
            printf("%d",a-b);
            break;
        case'*':
            printf("%d",a*b);
            break;
        case'/':
            if (b==0) {
                printf("Error! division by zero");
            }
            else {
                printf("%d",a/b);
            }
            break;
        default:
            printf("Invalid operator");
    }
    return 0;
}
